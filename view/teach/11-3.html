4 选择

4.1 if

// if (语句)
//   如果语句返回值不为0执行这里
// else
//   如果语句返回值为0执行这里

if (1)
    cout << "执行这里" << endl;
else
    cout << "不会执行这里" << endl;

if (0)
    cout << "不执行这里" << endl;
else
    cout << "执行这里" << endl;

if (1 == 1)
    cout << "当然了" << endl;
else
    cout << "不可能" << endl;

4.2 switch

switch(1) {  // 根据括号内的值匹配下列的case语句
    case 1:
        cout << "会输出这句话" << endl;
        break;  // 如果不加break会继续执行下面的语句
    case 2:
        cout << "不会执行" << endl;
        break;
    case 3:
        cout << "不会执行" << endl;
        break;
    default: cout << "全都不匹配才执行这里" << endl;
}

5 循环

5.1 while

while(1) {
    cout << "执行本句直到括号内的值为0" << endl;
}

// 从10输出到1
int a = 10;
while (a > 0) {
    cout << a << endl;
    a--;
}

5.2 for

// for(初始化;返回值为0结束否则继续;每次执行循环体结束后执行的语句)
//     循环体

// 从0输出到9
for (int i = 0; i < 10; i++)
    cout << i << endl;

5.3 continue

// continue的意思是停止本次循环，直接进行下一次循环

// 输出0至10的奇数
for (int i = 0; i < 10; i++)
    if (i % 2)
        cout << i << endl;
    else
        continue;

5.4 break

// break的意思是跳出整个循环

// 输出0-10的第一个奇数
for (int i = 0; i < 10; i++)
    if (i % 2)
        cout << i << endl;
    else
        break;

6 练习 2

输入一个数 n，输出 n!

6.1 题解

#include <iostream>

using namespace std;

int main() {
    int n;
    cin >> n;
    long tot = 1;
    for (int i = 1; i <= n; i++)
        tot *= i;
    cout << tot << endl;
    return 0;
}

7 数组

7.1 定义

// 类型名 数组名[数组长度]
int a[10]; // 定义了一个长度为10的整型数组a

// 通过 数组名[下标] 的方式找到第 下标+1 个元素
// 和变量的使用方法一样
a[0] = 0; // 给a数组的第一个元素赋值0

7.2 多维数组

// 有几个[]就有多少个维度
int a[10][2][3];  // 声明了一个三维的数组，总共有 10 * 2 * 3 个元素
                // 可以理解成有每个三维数组里有2个二维数组，每个二维数组里有3个一维数组
a[0][1][2] = 1; // 一样可以通过下标找到对应的元素

7.3 例题

输入一个字符串，把字符串逆序输出 

#include <iostream>

using namespace std;

int main() {
    int len = 0;
    cin >> len;
    char s[len];
    for (int i = 0; i < len; i++)
        cin >> s[i];
    for (int i = len - 1; i >= 0; i--)
        cout << s[i];
    cout << endl;
    return 0;
}
